# Overview

This hook is a specialized form of `useBpmnJsModeler` that uses the `BaseModeler` from `bpmn-js`. Similarly, it returns a callback which can be passed as `ref` property and the `BaseModeler` instance.

# Usage

```tsx
import React from "react";

import BaseModeler from "bpmn-js/lib/BaseModeler";
import { useBaseModeler, BaseViewerOptions } from "bpmn-react/Modeler";

const BpmnJsModeler = ({ xml }: { xml: string }) => {
    const [handleModelerRef, modeler] = useBaseModeler({} as BaseViewerOptions);

    return <div ref={ handleModelerRef } />
}
```
